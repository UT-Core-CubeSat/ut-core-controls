cmake_minimum_required(VERSION 3.10)

# Project Name
project(SOLACE_Sim_C++)

# Set C++ Standard (C++17 is required for your inline variables and clamps)
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# --------------------------------------------------------------------------
# INCLUDE DIRECTORIES
# --------------------------------------------------------------------------
# C++ folder (for local headers like Math.hpp, Parameters.hpp)
include_directories(${CMAKE_CURRENT_SOURCE_DIR})

# ADCSCore folder (for #include <ADCSCore/ADCSCore.hpp>)
include_directories(${CMAKE_CURRENT_SOURCE_DIR}/..)

# ADCSCore/components (for headers included by ADCSCore)
include_directories(${CMAKE_CURRENT_SOURCE_DIR}/../ADCSCore/components)

# --------------------------------------------------------------------------
# SOURCE FILES
# --------------------------------------------------------------------------
# Plant simulation (shared between both executables)
set(PLANT_SOURCES
    Dynamics.cpp
    Sensors.cpp
    two_body.cpp
    HelperFunctions.cpp
)

# Original simulation (uses C++ folder's Observer, Controller, ReferenceGenerator)
set(OLD_SIM_SOURCES
    main.cpp
    Observer.cpp
    ControllerManager.cpp
    ControllerNDI.cpp
    ControllerBDot.cpp
    ReferenceGenerator.cpp
)

# ADCSCore library sources
set(ADCSCORE_SOURCES
    ../ADCSCore/ADCSCore.cpp
    ../ADCSCore/components/Observer.cpp
    ../ADCSCore/components/ControllerManager.cpp
    ../ADCSCore/components/ControllerNDI.cpp
    ../ADCSCore/components/ControllerBDot.cpp
    ../ADCSCore/components/ReferenceGenerator.cpp
    ../ADCSCore/components/HelperFunctions.cpp
)

# New ADCSCore-based simulation
set(CORE_SIM_SOURCES
    main_core.cpp
)

# --------------------------------------------------------------------------
# EXECUTABLES
# --------------------------------------------------------------------------
# Original simulation (for comparison/regression testing)
add_executable(Sim ${PLANT_SOURCES} ${OLD_SIM_SOURCES})

# New ADCSCore-based simulation (the "harness")
add_executable(SimCore ${PLANT_SOURCES} ${CORE_SIM_SOURCES} ${ADCSCORE_SOURCES})

# --------------------------------------------------------------------------
# COMPILER OPTIONS
# --------------------------------------------------------------------------
if(MSVC)
    target_compile_options(Sim PRIVATE /MP /W3)
    target_compile_options(SimCore PRIVATE /MP /W3)
else()
    target_compile_options(Sim PRIVATE -Wall -O2)
    target_compile_options(SimCore PRIVATE -Wall -O2)
endif()
